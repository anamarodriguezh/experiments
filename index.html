                        baseEulers[idx * 3 + 1] = Math.cos(angle) * tilt;

                        startScales[idx] = 0;

                        targetScales[idx] = 0;

            let idx = 0;

            let textRegionSize = 65; // Default mapping size for desktop

            if (aspect < 0.6) textRegionSize = 25; // Scaled down further to prevent cropping while zoomed in

            else if (aspect < 0.9) textRegionSize = 40; // Tablet/Landscape mobile

            // Capture elapsed time from the PREVIOUS transition for smooth handoff

            const prevElapsed = performance.now() - transitionStartTime;

            for (let r = 0; r < gridRows; r++) {

                for (let c = 0; c < gridCols; c++) {

                    const worldX = (c / gridCols - 0.5) * groundWidth;

                    const worldY = -(r / gridRows - 0.5) * groundHeight;

                    const canvasX = ((worldX / textRegionSize) + 0.5) * canvasSize;

                    const canvasY = ((-worldY / textRegionSize) + 0.5) * canvasSize;

                    let isVein = false;

                    let haloIntensity = 0;

                    // Only map height/veins if within the central text area boundaries

                    if (canvasX >= 0 && canvasX < canvasSize && canvasY >= 0 && canvasY < canvasSize) {

                        const pixelIndex = (Math.floor(canvasY) * canvasSize + Math.floor(canvasX)) * 4;

                        const rVal = imgData[pixelIndex];

                        const bVal = imgData[pixelIndex + 2];

                        isVein = rVal > 100;

                        haloIntensity = bVal / 255;

                    }

                    const dx = worldX;

                    const dy = worldY;

                    const distFromCenter = Math.sqrt(dx * dx + dy * dy);

                    const baseDelay = distFromCenter * 4.0;

                    for (let s = 0; s < stalksPerCell; s++) {

                        if (idx >= totalCount) break;

                        // Compute current interpolated height (matches GPU shader's cubic ease-out)

                        const progress = Math.min(1, Math.max(0, (prevElapsed - delays[idx]) / 2000));

                        const easeFactor = 1 - Math.pow(1 - progress, 3);

                        const currentHeight = startScales[idx] + (targetScales[idx] - startScales[idx]) * easeFactor;

                        startScales[idx] = currentHeight;

                        let targetHeight = 0;

                        if (isVein) {

                            // Uniform, extremely flat peak height so the number acts as a solid, perfectly legible plateau

                            targetHeight = 4.0 + (Math.random() * 0.1);

                        } else if (haloIntensity > 0.02) {

                            const t = Math.pow(haloIntensity, 0.4);

                            targetHeight = 0.5 + t * 1.5 + Math.random() * 0.2;

                        } else {

                            // Extremely low, tiny background shards to isolate the number perfectly

                            targetHeight = 0.05 + Math.random() * 0.05;

                            if (currentHeight > 1.0) {

                                targetHeight = 0.1; // Clean reset

                            }

                        }

                        targetScales[idx] = targetHeight;

                        delays[idx] = baseDelay + Math.random() * 500;

                        idx++;

                    }

                }

            }

            if (stalkMesh && tipMesh) {

                stalkMesh.geometry.attributes.aStartScale.needsUpdate = true;

                stalkMesh.geometry.attributes.aTargetScale.needsUpdate = true;

                stalkMesh.geometry.attributes.aDelay.needsUpdate = true;

            }

            transitionStartTime = performance.now();

        }

                let weightedAvgX = 0;

                let weightedAvgY = 0;

                let totalWeight = 0;

                const maxR = Math.min(gridRows - 1, Math.ceil(mouseCellR + cellRadiusR));

                const minC = Math.max(0, Math.floor(mouseCellC - cellRadiusC));

                const maxC = Math.min(gridCols - 1, Math.ceil(mouseCellC + cellRadiusC));

                    for (let c = minC; c <= maxC; c++) {

                        const cellBaseIdx = (r * gridCols + c) * stalksPerCell;

                                }

        function initAudio() {

            if (audioCtx) {

                if (audioCtx.state === 'suspended') audioCtx.resume();

                return;

            }

            audioCtx = new (window.AudioContext || window.webkitAudioContext)();

            if (audioCtx.state === 'suspended') audioCtx.resume();

            // Add a Master Compressor to prevent any hard clipping/distortion

            const compressor = audioCtx.createDynamicsCompressor();

            compressor.threshold.value = -12; // Catch peaks earlier (was -6)

            compressor.knee.value = 12;

            compressor.ratio.value = 12; // Harder limiting (was 8)

            compressor.attack.value = 0.005; // Faster attack to stop transients (was 0.01)

            compressor.release.value = 0.25;

            compressor.connect(audioCtx.destination);

            // Add a High-pass filter to aggressively cut any remaining sub-bass (< 100Hz)

            const masterHighpass = audioCtx.createBiquadFilter();

            masterHighpass.type = 'highpass';

            masterHighpass.frequency.value = isMobile ? 180 : 120; // Raise cutoff for mobile speakers

            masterHighpass.Q.value = 0.7; // Standard smooth resonance

            masterHighpass.connect(compressor);

            masterGain = audioCtx.createGain();

            masterGain.gain.value = isMobile ? 0.4 : 0.5; // Lower overall headroom to prevent clipping

            masterGain.connect(masterHighpass); // Connect to highpass instead of compressor

            // Create a much lusher, ambient, cavernous delay

            masterDelay = audioCtx.createDelay();

            masterDelay.delayTime.value = 0.85; // Slower, wider delay

            delayFeedback = audioCtx.createGain();

            delayFeedback.gain.value = 0.75; // Longer, sustaining feedback

            const delayFilter = audioCtx.createBiquadFilter();

            delayFilter.type = 'lowpass';

            delayFilter.frequency.value = 800; // Warmer, darker echoes

            masterDelay.connect(delayFeedback);

            delayFeedback.connect(delayFilter);

            delayFilter.connect(masterDelay);

            masterDelay.connect(masterGain);

            // SUBTLE AUDIO UPDATE: Swapped back to gentle Sine waves for the ambient drone

            droneOsc = audioCtx.createOscillator();

            droneOsc.type = 'sine';

            const oscGain1 = audioCtx.createGain(); oscGain1.gain.value = 0.35;

            droneOsc2 = audioCtx.createOscillator();

            droneOsc2.type = 'sine';

            droneOsc2.detune.value = 12; // Gentle detune

            const oscGain2 = audioCtx.createGain(); oscGain2.gain.value = 0.25;

            droneGain = audioCtx.createGain();

            droneGain.gain.value = 0;

            droneFilter = audioCtx.createBiquadFilter();

            droneFilter.type = 'lowpass';

            droneFilter.frequency.value = 250;

            droneOsc.connect(oscGain1); oscGain1.connect(droneFilter);

            droneOsc2.connect(oscGain2); oscGain2.connect(droneFilter);

            droneFilter.connect(droneGain);

            droneGain.connect(masterGain);

            droneGain.connect(masterDelay);

            droneOsc.start();

            droneOsc2.start();

            audioEnabled = true;

        }

        function playPluck(frequency, brightness, waveformType, panValue = 0, duration = 2.5) {

            // Strict voice limit

            if (!audioEnabled || activePlucks >= maxVoices || isMuted) return;

            activePlucks++;

            // Trigger visual feedback on HUD (using cached element)

            if (hudPlucksEl) {

                hudPlucksEl.innerText = 'ACTIVE';

                hudPlucksEl.style.color = '#fff';

                clearTimeout(pluckTimer);

                pluckTimer = setTimeout(() => {

                    hudPlucksEl.innerText = 'IDLE';

                    hudPlucksEl.style.color = '#555';

                }, 150);

            }

            const osc = audioCtx.createOscillator();

            const gain = audioCtx.createGain();

            const filter = audioCtx.createBiquadFilter();

            let panner;

            if (audioCtx.createStereoPanner) {

                panner = audioCtx.createStereoPanner();

                panner.pan.value = panValue;

            } else {

                panner = audioCtx.createGain();

            }

            // Synthesizer waveform (square or sawtooth for cyberpunk edge)

            osc.type = waveformType;

            // Random chance for "sparkle" octaves when hitting spores hard/fast

            if (duration < 0.4 && Math.random() > 0.6) {

                frequency *= 2;

            }

            // PIAZZOLLA EFFECT: Passionate "glissando" bend on sustained notes (like a violin or bandoneon slide)

            if (Math.random() > 0.8 && duration > 0.5) {

                osc.frequency.setValueAtTime(frequency * 0.85, audioCtx.currentTime); // Start slightly flat

                osc.frequency.exponentialRampToValueAtTime(frequency, audioCtx.currentTime + 0.15); // Slide up sharply

            } else {

                osc.frequency.value = frequency;

            }

            // Use an envelope on the filter to simulate a synth pluck

            filter.type = 'lowpass';

            // SUBTLE AUDIO UPDATE: Heavily clamped brightness for soft, muted tones

            const clampedBrightness = Math.min(brightness, isMobile ? 1500 : 2500);

            filter.frequency.setValueAtTime(clampedBrightness * 0.8, audioCtx.currentTime);

            filter.frequency.exponentialRampToValueAtTime(Math.max(100, clampedBrightness * 0.1), audioCtx.currentTime + duration * 1.5);

            filter.Q.value = 1.2; // Softer resonance

            osc.connect(filter);

            filter.connect(gain);

            gain.connect(panner);

            panner.connect(masterDelay);

            panner.connect(masterGain);

            osc.start();

            // Dynamic, percussive, natural envelope

            gain.gain.setValueAtTime(0, audioCtx.currentTime);

            // SUBTLE AUDIO UPDATE: Soft attack fade-in and much longer, gentler fade-out

            const peakGain = Math.min(0.035, 0.015 + (brightness / 80000));

            gain.gain.linearRampToValueAtTime(peakGain, audioCtx.currentTime + 0.35); // Soft 350ms fade in

            gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + duration * 2.0); // Very long fade out

            osc.stop(audioCtx.currentTime + duration * 2.0);

            // Free up the voice when done

            osc.onended = () => {

                activePlucks--;

                osc.disconnect(); filter.disconnect(); gain.disconnect(); panner.disconnect();

            };

        }

        function playDustSound(intensity, speed, panValue) {

            if (!audioEnabled || activePlucks >= maxVoices || isMuted) return; // Strict voice limit

            if (Math.random() > 0.45) return; // Prevent triggering too frequently

            activePlucks++;

            const osc = audioCtx.createOscillator();

            const gain = audioCtx.createGain();

            const filter = audioCtx.createBiquadFilter();

            let panner;

            if (audioCtx.createStereoPanner) {

                panner = audioCtx.createStereoPanner();

                panner.pan.value = panValue;

            } else {

                panner = audioCtx.createGain();

            }

            // 8-bit style triangle chime

            osc.type = 'triangle';

            // Strictly snap dust sounds to the harmonious pentatonic scale instead of chaotic noise

            const scale = currentAudioProfile ? currentAudioProfile.scale : [261.63, 293.66, 329.63, 392.00, 440.00];

            const overtoneFreq = scale[Math.floor(Math.random() * scale.length)];

            osc.frequency.value = overtoneFreq * 2;

            filter.type = 'bandpass';

            filter.frequency.value = overtoneFreq * 2;

            filter.Q.value = 4;

            osc.connect(filter);

            filter.connect(gain);

            gain.connect(panner);

            panner.connect(masterDelay);

            panner.connect(masterGain);

            osc.start();

            gain.gain.setValueAtTime(0, audioCtx.currentTime);

            // Soft, glowing attack and very long sustained decay

            const peakGain = Math.min(0.02, intensity * 0.006 + 0.002);

            gain.gain.linearRampToValueAtTime(peakGain, audioCtx.currentTime + 0.1);

            gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + 1.5 + Math.random() * 1.5);

            osc.stop(audioCtx.currentTime + 3.0);

            osc.onended = () => {

                activePlucks--;

                osc.disconnect(); filter.disconnect(); gain.disconnect(); panner.disconnect();

            };

        }

        // Gracefully silence audio on page close to prevent fizz/pop

        window.addEventListener('beforeunload', () => {

            if (masterGain && audioCtx) {

                masterGain.gain.setValueAtTime(0, audioCtx.currentTime);

            }

            if (audioCtx) {

                audioCtx.close();

            }

        });
